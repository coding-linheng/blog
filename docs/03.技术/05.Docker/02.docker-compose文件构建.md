# docker-compose yaml文件构建


## 示例

### 基于现有镜像创建服务

```yaml
version: "3.8"

services:
  flask-demo:
    image: flask-demo:latest
    environment:
      - REDIS_HOST=redis-server
    networks:
      - demo-network
    ports:
      - 5000:5000

  redis-server:
    image: redis:latest
    networks:
      - demo-network

networks:
  demo-network:
```

### 构建镜像再创建服务容器

```yaml
version: "3.8" #docker-compose语法版本

services:  #镜像服务
flask-demo:  #服务名称
   build:    
      context: ../demo  #构建地址
      dockerfile: Dockerfile-build  #构建文件
   image: flask-demo:latest  #构建的镜像
   environment:  #环境变量
      - REDIS_HOST=redis-server
   networks:  #容器网卡
      - demo-network
   ports:  #端口映射
      - 5000:5000
      - 00:00  # 可指定多个

redis-server:
   image: redis:latest
   networks:
      - demo-network

busybox:
   image: busybox:latest
   command: ["sleep", "3600"]  #执行命令
   networks:
      - demo-network

networks:  #创建网卡
demo-network:
```

### networks

```yaml
version: "3.8"

services:
  box1:
    image: busybox:latest
    command: ["sleep", "3600"]
    networks:
      - demo-network

  box2:
    image: busybox:latest
    command: ["sleep", "3600"]
    networks:
      - demo-network
      - demo-network2


networks:
  demo-network:
    ipam:
      driver: default # 指定网卡类型 默认brdige
      config:  #网卡配置
        - subnet: 172.28.0.0/16 #网卡网段
  demo-network2:
    ipam:
      driver: default
```


### 水平扩展(--scale)

```yaml
version: "3.8"

services:
  flask:
    build:
      context: ../
      dockerfile: Dockerfile-build
    image: flask-demo:latest
    environment:
        - REDIS_HOST=redis-server
    networks:
        - redis-network
        - nginx-network

  redis-server:
    image: redis:latest
    networks:
        - redis-network

  nginx:
    image: nginx:stable-alpine
    ports:
        - 8000:80
    depends_on:
      - flask
    volumes:
        - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro
        - ./log/nginx:/var/log/nginx
    networks:
        - nginx-network


networks:
    redis-network:
    nginx-network:
```

运行命令行
```bash
docker-compose up -d --scale flask=3 --remove-orphans --build
```

创建三个flask服务用nginx做负载均衡


### 环境变量(environment)

```yaml
  redis-server:
    image: redis:latest
    command: redis-server --requirepass ${REDIS_PASSWORD} #密码设置环境变量
    networks:
        - redis-network
```

创建.env文件用于配置环境变量
```env
REDIS_PASSWORD=redis_password
```

运行docker-compose config查看当前配置
